function doPost(e) { try { let data = {}; if (e.postData && e.postData.type === 'application/json') { data = JSON.parse(e.postData.contents); } else { data = e.parameter; } const now = new Date(); const timestamp = Utilities.formatDate(now, 'Asia/Taipei', 'yyyy-MM-dd HH:mm:ss'); // ✨ 處理回報資料 (加入防呆與字數限制) if (data.action === 'report') { const reportSheet = SpreadsheetApp.getActive().getSheetByName('回報'); if (!reportSheet) throw new Error('找不到名為「回報」的工作表'); reportSheet.appendRow([ (data.brandName || '').substring(0, 100), // 限制 100 字 (data.reportType || '').substring(0, 50), // 限制 50 字 (data.details || '').substring(0, 1000), // 限制 1000 字 timestamp ]); return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON); } // --- 處理新增品牌 (加入防呆與字數限制) --- const sheet = SpreadsheetApp.getActive().getSheetByName('待審核'); if (!sheet) throw new Error('找不到名為「待審核」的工作表'); let logoUrl = ''; const link = (data.link || '').trim(); const safeBrandName = (data.brandName || 'MIT').substring(0, 100); // ✨ 網址驗證收緊：必須是真正的 http/https 格式 if (link && /^https?:\/\/.+/i.test(link)) { try { const domain = link.replace(/^(?:https?:\/\/)?(?:www\.)?/i, "").split('/')[0]; if (domain) { logoUrl = 'https://logo.clearbit.com/' + domain; } } catch(err) { logoUrl = ''; } } // 強制置頂插入法，並加入全欄位防護罩 sheet.insertRowAfter(1); const newRowData = [ safeBrandName, (data.category || '').substring(0, 50), (data.origin || '').substring(0, 100), (data.mit || '否').substring(0, 20), link.substring(0, 500), // 避免超長惡意網址 (data.desc || '').substring(0, 1000), // 介紹限制 1000 字 '待確認', logoUrl, (data.type || '消費者推薦').substring(0, 50), timestamp ]; sheet.getRange(2, 1, 1, 10).setValues([newRowData]); // 神奇魔法：使用強制資料驗證生成核取方塊 const checkboxRule = SpreadsheetApp.newDataValidation().requireCheckbox().build(); sheet.getRange(2, 11).setDataValidation(checkboxRule); sheet.getRange(2, 11).setValue('FALSE'); return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON); } catch (err) { console.error("後台解析發生錯誤：", err.toString()); return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: 'ok' })).setMimeType(ContentService.MimeType.JSON); } // --- 以下為業主專用：核准上架與網址沙盒防護 --- function onEdit(e) { if (!e || !e.range) return; const sheet = e.source.getActiveSheet(); if (sheet.getName() !== '待審核' || e.range.getColumn() !== 11) return; if (e.value !== 'TRUE') return; const row = e.range.getRow(); if (row === 1) return; const rowDataRange = sheet.getRange(row, 1, 1, 10); const rowData = rowDataRange.getValues()[0]; const targetUrl = rowData[4]; // 取得目前打開這份試算表的使用者介面 (用來顯示優雅的 Toast 通知) const ui = SpreadsheetApp.getUi(); const safetyCheck = verifyUrlSandbox(targetUrl); if (!safetyCheck.isSafe) { e.range.setValue('FALSE'); sheet.getRange(row, 1, 1, 11).setBackground('#ffe6e6'); // 🛡️ 升級 1：改用無阻塞的 Toast 通知警告 ui.alert('⚠️ 系統攔截', '該連結具有風險：\n\n' + safetyCheck.reason, ui.ButtonSet.OK); return; } rowData[6] = '已確認'; const liveSheet = e.source.getSheetByName('工作表1'); if (liveSheet) { liveSheet.appendRow(rowData); sheet.deleteRow(row); // 🛡️ 升級 2：改用右下角 Toast，優雅且不中斷程式 e.source.toast('網址驗證通過，已自動搬移至正式表單！', '✅ 上架成功', 3); } else { ui.alert('❌ 錯誤', '找不到名為「工作表1」的正式表單。', ui.ButtonSet.OK); } } function verifyUrlSandbox(url) { if (!url || url.trim() === '') return { isSafe: true }; if (!/^https?:\/\/.+/i.test(url)) return { isSafe: false, reason: '網址格式錯誤' }; const blacklist = ['scam', 'free-gift', 'login-verify', 'bit.ly', 'is.gd', 'j.mp', 'reurl.cc']; for (let i = 0; i < blacklist.length; i++) { if (url.toLowerCase().includes(blacklist[i])) return { isSafe: false, reason: '觸發短網址或高風險詐騙字詞防護' }; } try { const response = UrlFetchApp.fetch(url, { muteHttpExceptions: true, followRedirects: false, timeout: 4000 }); const statusCode = response.getResponseCode(); if (statusCode === 404) return { isSafe: false, reason: '網站已失效 (404 Not Found)' }; if (statusCode === 301 || statusCode === 302) { const redirectUrl = response.getHeaders()['Location']; if (redirectUrl && !redirectUrl.startsWith('http')) return { isSafe: false, reason: '偵測到異常的隱藏轉址行為' }; } return { isSafe: true }; } catch (err) { return { isSafe: true }; } }
